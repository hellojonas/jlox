expression				::= equality ;
equality				::= comparison ( (  "!=" | "==" ) comparison )* ;
comparison				::= term ( (  "<" | "<=" | ">" | ">=" ) term )* ;
term					::= factor ( (  "+" | "+" ) factor )* ;
factor					::= unary ( (  "*" | "/" ) unary )* ;
unary					::= ( "!" | "-" ) unary 
							| primary ;
primary					::= "true" | "false" | "nil"
							| NUMBER
							| STRING
							| IDENTIFIER 
							| "(" expression ")";
